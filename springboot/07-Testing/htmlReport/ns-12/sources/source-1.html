  *
  * @param <T>  Tipo de dato
  * @param <ID> Tipo de dato del ID
  */

     Optional<T> findById(ID id);
 
     Boolean existsById(ID id);
 
     T save(T t);
 
     void deleteById(ID id);
 
     void delete(T t);
 
     long count();
 
     void deleteAll();
 }
